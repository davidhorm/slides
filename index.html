Make Veggie ðŸ¥ª: FP Example


					// ['Loaf of Bread', 'Loaf of Bread', 'Slab Bacon', 'Whole Tomato', 'Head of Lettuce']
					// => 'The sandwich has: slice of Bread, sliced Tomato, and Lettuce leaf, and slice of Bread'
					const makeVeggieSandwich = (ingredients: string[]) => {
						const sandwich = ingredients
							.filter((ingredient) => ingredient.type !== 'meat')
							.map(getSlicedIngredient)
							.sort(sortIngredients)
							.reduce(
								(sandwich, ingredient) => `${sandwich}${ingredient.value}, and `,
								'The sandwich has: '
							);
						
						return sandwich;
					};
